535/* Naive loop; used when perf.start/stop() is expected to be slow or imprecise, or when
536 * we have no ASM cycle counters, or when the number of iterations is low */
537#define CHECKASM_PERF_BENCH_SIMPLE(count, time, ...) \
538 do { \
539 time = perf.start(); \
540 for (int tidx = 0; tidx < count; tidx++) \
541 bench_func(__VA_ARGS__); \
542 time = perf.stop(time); \
543 } while (0)
544
545/* Unrolled loop with inline outlier rejection; used when we have asm cycle counters */
546#define CHECKASM_PERF_BENCH_ASM(total_count, time, ...) \
547 do { \
548 int tcount_trim = 0; \
549 uint64_t tsum_trim = 0; \
550 for (int titer = 0; titer < total_count; titer += 32) { \
551 uint64_t t = CHECKASM_PERF_ASM(); \
552 CHECKASM_PERF_CALL16(__VA_ARGS__); \
553 CHECKASM_PERF_CALL16(__VA_ARGS__); \
554 t = CHECKASM_PERF_ASM() - t; \
555 if (t * tcount_trim <= tsum_trim * 4 && (titer > 0 || total_count < 1000)) { \
556 tsum_trim += t; \
557 tcount_trim++; \
558 } \
559 } \
560 time = tsum_trim; \
561 total_count = tcount_trim << 5; \
562 } while (0)
